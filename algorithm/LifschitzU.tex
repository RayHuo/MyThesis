\begin{algorithm}\label{LifschitzU}
	\SetKwData{Dis}{dis}
	\SetKwFunction{CalculateHammingDistance}{CalculateHammingDistance}
	\SetKw{downto}{downto}\SetKw{return}{return}\SetKw{Break}{break}
	\SetKwInOut{Input}{输入}\SetKwInOut{Output}{输出}
	\caption{计算分割集的算法$U(P)$}
	\Input{一个逻辑程序$P$}
	\Output{分割集$U$}
	
	$U~:=~\emptyset$；      \\
	$r~:=~U$中的第一个规则；	\\
	$U~:=~U \cup Atoms(r)$；	\\
	$a~:=~U$中的第一个原子；\\
	\While {$U~is~changed$} {
		$rules~:=~\{r \in P | head(r) \cap a \neq \emptyset \}$ \\
		\For {$rule \in rules$} {
			$U~:=~U \cup Atoms(rule)$
		}
		$a~:=~U$中的下一个原子；\\
	}
	
	\return $U$
\end{algorithm}
