\begin{algorithm}\label{LifschitzU}
	\SetKwData{Dis}{dis}
	\SetKwFunction{CalculateHammingDistance}{CalculateHammingDistance}
	\SetKw{downto}{downto}\SetKw{return}{return}\SetKw{Break}{break}
	\SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
	\caption{计算所有分割集的算法$Us(P)$}
	\Input{An ASP logic program $P$}
	\Output{A set of all splitting sets $Us$}
	
	$Us~:=~\emptyset$;	\\
	\For { $rule \in P$ } {
		$U~:=~Atoms(rule)$;	\\
		$P'~:=~P \backslash \{ rule \}$;	\\
		\For {$r \in P'$} {
			\If {$head(r) \cap U \neq \emptyset$} {
				$U~:=~U \cup Atoms(r)$;	\\
			}	
		}	
		$Us~:=~Us \cup \{ U \}$;	\\
	}
	
	\return $Us$
