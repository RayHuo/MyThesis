\begin{algorithm}\label{UP}
	\SetKwData{Dis}{dis}
	\SetKwFunction{CalculateHammingDistance}{CalculateHammingDistance}
	\SetKw{downto}{downto}\SetKw{return}{return}\SetKw{Break}{break}
	\SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
	\caption{计算单位传播的算法$UP(P)$}
	\Input{An ASP logic program $P$}
	\Output{A set of literals $lit$}
	
	$\Gamma~:=$ clauses of $P$;	\\
	\If {$\emptyset \in \Gamma$} {
		\return $Lit(P)$;
	}
	$A~:=~unit\_clause(\Gamma)$;	\\
	\If {$A$ is inconsistent} {
		\return $Lit(P)$;	\\
	}
	\If {$A \neq \emptyset$} {
		\return $A \cup UP(assign(A, \Gamma))$;	\\
	}
	\Else {
		\return $\emptyset$;	
	}

	
\end{algorithm}
